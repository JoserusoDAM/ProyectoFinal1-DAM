DELIMITER $$
CREATE DEFINER=`JoseA`@`%` PROCEDURE `insertClub`(IN `vid_club` INT, IN `vnombre` VARCHAR(60), IN `vfecha_creacion` INT, IN `vnom_estadio` VARCHAR(60))
    NO SQL
BEGIN
INSERT INTO Clubs
(id_club, nombre, fecha_creacion, nom_estadio) 
VALUES
(vid_club, vnombre, vfecha_creacion, vnom_estadio);
END$$
DELIMITER ;


DELIMITER $$
CREATE DEFINER=`JoseA`@`%` PROCEDURE `deleteClub`(IN `vclub` INT)
    NO SQL
BEGIN 
DELETE FROM Clubs WHERE id_club = vclub;
END$$
DELIMITER ;


DELIMITER $$
CREATE DEFINER=`JoseA`@`%` PROCEDURE `updateClub`(IN `vid_club` INT, IN `vnombre` VARCHAR(60), IN `vfecha_creacion` INT, IN `vnom_estadio` VARCHAR(60))
    NO SQL
BEGIN
UPDATE Clubs SET nombre = vnombre, fecha_creacion = vfecha_creacion,
nom_estadio = vnom_estadio WHERE id_club = vid_club;
END$$
DELIMITER ;
