CREACION

CREATE TABLE Clubs (
id_club INT (11) NOT NULL AUTO_INCREMENT,
nombre VARCHAR (100) NOT NULL UNIQUE,
fecha_creacion INT (11),
nom_estadio VARCHAR (100),
PRIMARY KEY (id_equipo, nombre)
)

CREATE TABLE Futbolistas(
id_futbolista INT (11) NOT NULL AUTO_INCREMENT,
nif VARCHAR (9) NOT NULL UNIQUE,
nombre VARCHAR (60),
apellidos VARCHAR (120),
fecha_nacimiento DATE,
nacionalidad VARCHAR (40),
PRIMARY KEY (id_futbolista, nif)
)

CREATE TABLE Historico(
id_club INT (11) NOT NULL,
id_futbolista INT (11) NOT NULL,
temporada INT (11) NOT NULL,
PRIMARY KEY (id_club, id_futbolista, temporada),
FOREIGN KEY (id_club) REFERENCES Clubs(id_club),
FOREIGN KEY (id_futbolista) REFERENCES Futbolistas(id_futbolista)
)
